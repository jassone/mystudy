## 值拷贝和值引用

## 一、几个demo

### 1、demo1

```go
type student struct {
	Name string
	Age  int
}
func main() {
   //定义map
   m := make(map[string]*student)

   //定义student数组
   stus := []student{
      {Name: "zhou", Age: 24},
      {Name: "li", Age: 23},
      {Name: "wang", Age: 22},
   }

   //将数组依次添加到map中
   for _, stu := range stus {
      m[stu.Name] = &stu // 如果这里改为值赋值，则就正常了
   }

   //打印map
   for k,v := range m {
      fmt.Println(k ,"=>", v.Name)
   }
   // map中的3个key均指向数组中最后一个结构体。
   //zhou => wang
   //li => wang
   //wang => wang

   // 原因
   // for range中，stu是结构体的一个拷贝副本，所以m[stu.Name]=&stu实际上一致指向同一个指针，
   // 即 for range 使用短变量声明(:=)的形式迭代变量时，变量 stu 在每次循环体中都会被重用，
   // 而不是重新声明。所以 stu 每次填充的都是临时变量 value 的地址，
   // 最终该指针的值为遍历的最后一个struct的值拷贝。
}
```

##### 正确写法

```go
// 正确写法
func main1() {
   //定义map
   m := make(map[string]*student)

   //定义student数组
   stus := []student{
      {Name: "zhou", Age: 24},
      {Name: "li", Age: 23},
      {Name: "wang", Age: 22},
   }

   // 注意这里：遍历结构体数组，依次赋值给map
   for i := 0; i < len(stus); i++  {
      m[stus[i].Name] = &stus[i]
   }

   //打印map
   for k,v := range m {
      fmt.Println(k ,"=>", v.Name)
   }
}

// 正确写法2
func main2() {
   //定义map
   m := make(map[string]*student)

   //定义student数组
   stus := []student{
      {Name: "zhou", Age: 24},
      {Name: "li", Age: 23},
      {Name: "wang", Age: 22},
   }

   //将数组依次添加到map中
   for _, stu := range stus {
      stutemp := stu // 这里会重新分配变量内存 todo???
      m[stu.Name] = &stutemp
   }
   // 或者
   for i, stu := range stus {
      m[stu.Name] = &stus[i]
   }

   //打印map
   for k,v := range m {
      fmt.Println(k ,"=>", v.Name)
   }
}
```

### 2、demo2

和上面类型

```go
func main33() {
   x := []int{0, 1, 2}
   y := [3]*int{}
   for i, v := range x { // for-range 虽然使用的是 :=，但是 v 不会重新声明
      defer func() {
         print(v) // 2 2 2
      }()
      y[i] = &v
   }
   print(*y[0], *y[1], *y[2]) // 2 2 2
}
```

正确做法

```go
func main() {
   x := []int{0, 1, 2}
   y := [3]int{}
   for i, v := range x { // for-range 虽然使用的是 :=，但是 v 不会重新声明
      defer func() {
         print(v) // 2 2 2
      }()
      y[i] = v
   }
   print(y[0], y[1], y[2]) // 0 1 2
}
```
